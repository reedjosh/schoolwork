\section*{Binomial Distribution}

\subsection*{General}
$X$ = the number of successes in n trials.

\subsection*{Mass Function}
$P\{X=k\} = \binom{n}{k}p^kq^{n-k}$, for $k=0,1,2,...,n,$ where $\binom{n}{k}=\frac{n!}{k!(n-k)!}$

\subsection*{Mean}
$m_x=np$

\subsection*{Variance}
$Var(x)=np(1-p)$
