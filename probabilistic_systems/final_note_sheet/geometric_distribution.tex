\section*{Geometric Distribution}

\subsection*{General}
$X$ is the number of trials before the first success \\
$p$ is the probability of success

\subsection*{Mass Function}
$P\{X=k\} = p(1-p)^{k-1}$, for $k=1,2,3,...$

\subsection*{Mean}
$m_x=1/p$

\subsection*{Variance}
$Var(X)=(1-p)/p^2$
